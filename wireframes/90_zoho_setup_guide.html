increment_review_count(applicationId : STRING)

// Fired by WR-C-3 when a Review Scores subform row is added.
appId   = applicationId.toLong();
app     = zoho.crm.getRecordById("Deals", appId);
current = ifnull(app.get("Review_Count"), 0).toLong();
upd     = Map();
upd.put("Review_Count", current + 1);
zoho.crm.updateRecord("Deals", appId, upd);
info "Review_Count -> " + (current + 1);
return "ok";

unlock_all_review_scores(applicationId : STRING)

// Fired by WR-C-4 when Review_Count ≥ 2. Flips Visible_to_NC_Admin true on every row.
appId = applicationId.toLong();
app   = zoho.crm.getRecordById("Deals", appId);
rows  = ifnull(app.get("Review_Scores"), List());
updated = List();
for each r in rows {
  r.put("Visible_to_NC_Admin", true);
  updated.add(r);
}
upd = Map();
upd.put("Review_Scores", updated);
zoho.crm.updateRecord("Deals", appId, upd);
return "ok";

create_safeguarding_case(applicationId : STRING, raisedByUserId : STRING)

// Fired by WR-C-5 when Safeguarding_Flag_Status changes from None.
appId      = applicationId.toLong();
app        = zoho.crm.getRecordById("Deals", appId);
contactId  = app.get("Contact_Name").get("id");
caseRec    = Map();
caseRec.put("Reporter_Lookup", raisedByUserId);
caseRec.put("Date_Raised", zoho.currenttime.toString("yyyy-MM-dd'T'HH:mm:ss"));
caseRec.put("SG_Status", "Under Review");
caseRec.put("Application_Lookup", appId);
caseRec.put("Contact_Lookup", contactId);
created = zoho.crm.createRecord("Safeguarding_Cases", caseRec);
return created.get("id");

Field-level security — blind review

Zoho FLS doesn't natively scope by row-value
Native FLS hides a field for a profile across all rows. To hide a peer reviewer's row scores but show your own, we use a Client Script on the Application detail page: wipes peer score cells unless All_Reviews_Complete = true. The script lives at zoho-build/client-scripts/blind_review.js (Priyanka builds it during Fri AM slot).

generate_nomination_pack(applicationId : STRING)

// Custom button. Builds UWC-branded HTML, creates Pack record, attaches file.
appId     = applicationId.toLong();
app       = zoho.crm.getRecordById("Deals", appId);
applicant = zoho.crm.getRecordById("Contacts", app.get("Contact_Name").get("id"));
school    = zoho.crm.getRecordById("Schools",  app.get("School_Nominated").get("id"));
nc        = zoho.crm.getRecordById("National_Committees", app.get("National_Committee").get("id"));

ncCode = ifnull(nc.get("NC_Code"), "XX");
yr     = zoho.currenttime.year();
suffix = app.get("Deal_Name").subString(app.get("Deal_Name").lastIndexOf("-") + 1);
ncRef  = ncCode + "-" + yr + "-" + suffix;

html = "<!doctype html><html><head><style>"
     + ".banner{background:#003087;color:#fff;padding:28px;font-size:22px;}"
     + ".section{padding:18px;border-bottom:1px solid #eee;}"
     + "</style></head><body>"
     + "<div class='banner'>UWC Nomination Pack — " + ncRef + "</div>"
     + "<div class='section'><b>Applicant</b> " + applicant.get("First_Name") + " " + applicant.get("Last_Name") + "</div>"
     + "<div class='section'><b>Nominating NC</b> " + nc.get("NC_Name") + "</div>"
     + "<div class='section'><b>Target School</b> " + school.get("School_Name") + "</div>"
     + "</body></html>";

pack = Map();
pack.put("Application_Lookup", appId);
pack.put("Target_School_Lookup", school.get("id"));
pack.put("Pack_Status", "Generated");
pack.put("Generated_At", zoho.currenttime.toString("yyyy-MM-dd'T'HH:mm:ss"));
pack.put("NC_Reference", ncRef);
created = zoho.crm.createRecord("Nomination_Packs", pack);
zoho.crm.attachFile("Nomination_Packs", created.get("id").toLong(), html);
return created.get("id");

Scheduled functions

recompute_competency_gaps() — Daily @ 02:45 UTC

comps  = zoho.crm.getRecords("Competencies", 1, 200);
bodies = zoho.crm.searchRecords("Governance_Bodies", "(Body_Type:equals:Board)and(Body_Status:equals:Active)");
for each c in comps {
  gap = 0; total = 0;
  for each b in bodies {
    crit = "(Governance_Body:equals:" + b.get("id") + ")and(Membership_Status:equals:Active)";
    members = zoho.crm.searchRecords("Governance_Memberships", crit);
    found = false;
    for each m in members {
      for each h in ifnull(m.get("Competencies_Held"), List()) {
        if (h.get("id") == c.get("id").toString()) { found = true; total = total + 1; }
      }
    }
    if (!found) gap = gap + 1;
  }
  zoho.crm.updateRecord("Competencies", c.get("id").toLong(), {"Gap_Count": gap, "Total_Holders": total});
}
